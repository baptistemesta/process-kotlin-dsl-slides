=== Kotlin makes writing DSL easy

Function literals with receiver
[source,kotlin]
----
fun html(init: HTML.() -> Unit): HTML {
    val html = HTML()
    html.init()
    return html
}
//gives
html {
 // ...
}
----

=== Kotlin makes writing DSL easy

infix
[source,kotlin]
----
infix fun Int.shl(x: Int): Int { ... }

// calling the function using the infix notation
1 shl 2

// is the same as
1.shl(2)
----
